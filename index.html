import streamlit as st import pandas as pd # إعدادات الصفحة st.set_page_config(page_title="نظام إدارة شراكة المشروع", layout="wide") st.title("📊 نظام ضبط المصاريف وتوزيع الأرباح") st.info("إدارة الشراكة بين 3 أشخاص - نسبة المدير 15% من المبيعات") # أسماء الشركاء partners = ["الشريك 1", "الشريك 2", "الشريك 3"] manager = "الشريك 1" # افترضنا أنك الشريك الأول # قاعدة بيانات وهمية (يمكن ربطها بـ Google Sheets لاحقاً) if 'expenses' not in st.session_state: st.session_state.expenses = [] if 'sales' not in st.session_state: st.session_state.sales = [] # القسم الأول: إضافة العمليات col1, col2 = st.columns(2) with col1: st.subheader("🛒 إضافة مصروف (فواتير، أعلاف، صيانة)") payer = st.selectbox("من الذي دفع؟", partners) amount = st.number_input("المبلغ المدفوع", min_value=0.0, step=1.0) note = st.text_input("بيان المصروف (مثلاً: فاتورة كهرباء)") if st.button("تسجيل المصروف"): st.session_state.expenses.append({"الشريك": payer, "المبلغ": amount, "البيان": note}) st.success("تم تسجيل المصروف") with col2: st.subheader("💰 تسجيل عملية بيع") sale_amount = st.number_input("إجمالي قيمة البيع", min_value=0.0, step=1.0) sale_note = st.text_input("بيان البيعة") if st.button("تسجيل البيع"): # حساب نسبة المدير 15% manager_fee = sale_amount * 0.15 remaining = sale_amount - manager_fee st.session_state.sales.append({ "الإجمالي": sale_amount, "عمولة المدير (15%)": manager_fee, "المتبقي للتوزيع": remaining, "البيان": sale_note }) st.success(f"تم التسجيل. عمولتك: {manager_fee} ريال") st.divider() # القسم الثاني: الحسابات الختامية st.header("📈 ملخص الحسابات") df_exp = pd.DataFrame(st.session_state.expenses) df_sales = pd.DataFrame(st.session_state.sales) if not df_exp.empty or not df_sales.empty: col_a, col_b, col_c = st.columns(3) total_spent = df_exp['المبلغ'].sum() if not df_exp.empty else 0 total_sales = df_sales['الإجمالي'].sum() if not df_sales.empty else 0 total_fees = df_sales['عمولة المدير (15%)'].sum() if not df_sales.empty else 0 col_a.metric("إجمالي المصاريف", f"{total_spent} ريال") col_b.metric("إجمالي المبيعات", f"{total_sales} ريال") col_c.metric("عمولتك المستحقة (15%)", f"{total_fees} ريال") # حساب مديونية كل شريك (Splitwise Style) st.subheader("⚖️ تسوية الشركاء (من له ومن عليه)") share_per_person = total_spent / 3 for p in partners: p_spent = df_exp[df_exp['الشريك'] == p]['المبلغ'].sum() if not df_exp.empty else 0 balance = p_spent - share_per_person if balance > 0: st.write(f"✅ **{p}**: دفع أكثر من حصته، يستحق استرداد: **{balance:.2f} ريال**") else: st.write(f"❌ **{p}**: دفع أقل من حصته، عليه دفع: **{abs(balance):.2f} ريال**") else: st.warning("لا توجد بيانات مسجلة بعد.")
